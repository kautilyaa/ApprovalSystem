CREATE TABLE `request`.`bill_submit` ( `USERNAME` VARCHAR(25) NOT NULL , `BILL_NO` INT NOT NULL AUTO_INCREMENT , `BILL_DETAILS` VARCHAR(100) NOT NULL , `MANAGER_NAME` VARCHAR(35) NOT NULL , `DEPT_TYPE` VARCHAR(20) NOT NULL , `NAME` VARCHAR(40) NOT NULL , PRIMARY KEY (`BILL_NO`)) ENGINE = InnoDB;

CREATE TABLE `request`.`bill_report` ( `USERNAME` VARCHAR(25) NOT NULL , `BILL_NUM` INT NOT NULL , `BILL_DETAILS` VARCHAR(100) NOT NULL , `DEPT_TYPE` VARCHAR(15) NOT NULL , `STATUS` VARCHAR(60) NOT NULL , PRIMARY KEY (`BILL_NUM`)) ENGINE = InnoDB;

CREATE TABLE `request`.`login` ( `NAME` VARCHAR(50) NOT NULL , `MOBILE_NO` INT NOT NULL , `EMAIL_ID` VARCHAR(100) NOT NULL , `USERNAME` VARCHAR(25) NOT NULL , `PASSWORD` VARCHAR(25) NOT NULL , `LOGIN_TYPE` VARCHAR(20) NOT NULL , `SECURITY_ANS` VARCHAR(35) NOT NULL , `DEPT_TYPE` VARCHAR(20) NULL , PRIMARY KEY (`USERNAME`)) ENGINE = InnoDB;

CREATE TABLE  `request`.`FINANCE_BILL` ( `USERNAME` VARCHAR(25) NOT NULL , 
	`BILL_NO` INT NOT NULL , 
	`BILL_DETAILS` VARCHAR(100) NOT NULL , 
	`MANAGER_NAME` VARCHAR(35) NOT NULL , 
	`DEPT_TYPE` VARCHAR2(15) NOT NULL , 
	`NAME` VARCHAR2(35) NOT NULL , 
	  UNIQUE (`BILL_NO`)) ENGINE = InnoDB;
